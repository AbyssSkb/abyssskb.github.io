常用不定积分表

∫kdx=kx+C\int kdx = kx + C∫kdx=kx+C

∫xμdx=xμ+1μ+1+C\int x^{\mu}dx = \frac{x^{\mu + 1}}{\mu + 1} + C∫xμdx=μ+1xμ+1​+C

∫dxx=ln⁡∣x∣+C\int\frac{dx}{x} = \ln\left\vert x\right\vert + C∫xdx​=ln∣x∣+C

∫dx1+x2=arctan⁡x+C\int\frac{dx}{1 + x^2} = \arctan x + C∫1+x2dx​=arctanx+C

∫dx1−x2=arcsin⁡x+C\int\frac{dx}{\sqrt{1 - x^2}} = \arcsin x + C∫1−x2​dx​=arcsinx+C

∫sin⁡xdx=−cos⁡x+C\int\sin xdx = -\cos x + C∫sinxdx=−cosx+C

∫cos⁡xdx=sin⁡x+C\int\cos xdx = \sin x + C∫cosxdx=sinx+C

∫tan⁡xdx=−ln⁡∣cos⁡x∣+C\int\tan xdx = -\ln\left\vert \cos x\right\vert + C∫tanxdx=−ln∣cosx∣+C

∫cot⁡xdx=ln⁡∣sin⁡x∣+C\int\cot xdx = \ln\left\vert\sin x\right\vert + C∫cotxdx=ln∣sinx∣+C

∫sec⁡xdx=ln⁡∣sec⁡x+tan⁡x∣+C\int\sec xdx = \ln\left\vert\sec x + \tan x\right\vert + C∫secxdx=ln∣secx+tanx∣+C

∫csc⁡xdx=ln⁡∣csc⁡x−cot⁡x∣+C\int\csc xdx = \ln\left\vert\csc x - \cot x\right\vert + C∫cscxdx=ln∣cscx−cotx∣+C

∫sec⁡2xdx=tan⁡x+C\int\sec^2 xdx = \tan x + C∫sec2xdx=tanx+C

∫csc⁡2xdx=−cot⁡x+C\int\csc^2 xdx = -\cot x + C∫csc2xdx=−cotx+C

∫sec⁡xtan⁡xdx=sec⁡x+C\int\sec x\tan xdx = \sec x + C∫secxtanxdx=secx+C

∫csc⁡xcot⁡xdx=−csc⁡x+C\int\csc x\cot xdx = -\csc x + C∫cscxcotxdx=−cscx+C

∫exdx=ex+C\int e^xdx = e^x + C∫exdx=ex+C

∫axdx=axln⁡a+C\int a^xdx = \frac{a^x}{\ln a} + C∫axdx=lnaax​+C

∫dxa2+x2=1aarctan⁡xa+C\int\frac{dx}{a^2 + x^2} = \frac{1}{a}\arctan{\frac{x}{a}} + C∫a2+x2dx​=a1​arctanax​+C

∫dxa2−x2=12aln⁡∣a+xa−x∣+C\int\frac{dx}{a^2 - x^2} = \frac{1}{2a}\ln\left\vert\frac{a + x}{a - x}\right\vert + C∫a2−x2dx​=2a1​ln​a−xa+x​​+C

∫dxa2−x2=arcsin⁡xa+C\int\frac{dx}{\sqrt{a^2 - x^2}} = \arcsin{\frac{x}{a}} + C∫a2−x2​dx​=arcsinax​+C

∫dxx2+a2=ln⁡(x+x2+a2)+C\int\frac{dx}{\sqrt{x^2 + a^2}} = \ln\left(x + \sqrt{x^2 + a^2}\right) + C∫x2+a2​dx​=ln(x+x2+a2​)+C

∫dxx2−a2=ln⁡∣x+x2−a2∣+C\int\frac{dx}{\sqrt{x^2 - a^2}} = \ln\left\vert x + \sqrt{x^2 - a^2}\right\vert + C∫x2−a2​dx​=ln​x+x2−a2​​+C
